9 Vector circumcenter(const Vector& v1,const Vector& v2,const Vector& v3) {
10  Vector v12 = v1 - v2;
11  Vector v13 = v1 - v3;
12  Vector v23 = v2 - v3;
13  Scalar d = 2 * magSq(v12 ^ v23);
14  Scalar a = magSq(v23) * (v12 & v13) / d;
15  Scalar b = magSq(v13) * (-v12 & v23) / d;
16  Scalar c = magSq(v12) * (v13 & v23) / d;
17  return a * v1 + b * v2 + c * v3;
18 }
