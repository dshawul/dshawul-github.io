86  kappa(0.41),
87  ks(0),
88  cks(0.5)
89  {
90  init();
91  }
92  void init() {
93  yLog = 11.3f;
94  for(Int i = 0;i < 20;i++)
95  yLog = log(E * yLog) / kappa;
96  }
97  Scalar getUstar(Scalar nu,Scalar U,Scalar y) {
98  Scalar a = kappa * U * y / nu;
99  Scalar yp = a;
100  for(Int i = 0;i < 10;i++)
101  yp = (a + yp) / (1 + log(E * yp));
102  Scalar ustar = yp * nu / y;
103  return ustar;
104  }
105  Scalar getUp(Scalar ustar,Scalar nu,Scalar yp) {
106  Scalar up,dB;
107  Scalar ksPlus = (ustar * ks) / nu;
108  if(ksPlus < 2.25) {
109  dB = 0;
110  } else if(ksPlus < 90) {
111  dB = (1 / kappa) * log((ksPlus - 2.25) / 87.75 + cks * ksPlus)
112  * sin(0.4258 * (log(ksPlus) - 0.811));
113  } else {
114  dB = (1 / kappa) * log(1 + cks * ksPlus);
115  }
116  if(yp > yLog) up = log(E * yp) / kappa - dB;
117  else up = yp;
118  return up;
119  }
