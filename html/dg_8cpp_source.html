33  Scalar a,b;
34  Scalar L2, L2_1, L2_2;
35  Scalar L1, L1_1, L1_2;
36  L1 = 0; L1_1 = 0; L1_2 = 0;
37  L0 = 1; L0_1 = 0; L0_2 = 0;
38 
39  for(int i = 1; i <= p; i++) {
40  L2 = L1; L2_1 = L1_1; L2_2 = L1_2;
41  L1 = L0; L1_1 = L0_1; L1_2 = L0_2;
42  a = (2 * i - 1.0) / i;
43  b = (i - 1.0) / i;
44  L0 = a * x * L1 - b * L2;
45  L0_1 = a * (L1 + x * L1_1) - b * L2_1;
46  L0_2 = a * (2 * L1_1 + x * L1_2) - b * L2_2;
47  }

432  // Part I
433  // ======
434  // Note that the Jacobian is stored inverted and transposed so that it
435  // can be used directly for gradient/divergence calculations.
436  //
437  // {x,y,z} = J * {a,b,c} where J is
438  //
439  // [dx/da dx/db dx/dc]
440  // J = [dy/da dy/db dy/dc]
441  // [dz/da dz/db dz/dc]
442  //
443  // Also, {a,b,c} = J' * {x,y,z} where J' is the inverse of J
444  //
445  // [da/dx da/dy da/dz]
446  // J' = [db/dx db/dy db/dz]
447  // [dc/dx dc/dy dc/dz]
448  //
449  // The transpose of J' is used to compute derivatives of q
450  //
451  // {dq/dx, dq/dy, dq/dz} = trn(J') * {dq/da, dq/db, dq/dc}
452  //
453  // While the transpose of J is used for the inverse

455  // {dq/da, dq/db, dq/dc} = trn(J) * {dq/dx, dq/dy, dq/dz}
