424  Int i;
425  /*allocate*/
426  mFC.assign(mFacets.size(),Vector(0));
427  mCC.assign(mCells.size(),Vector(0));
428  mFN.assign(mFacets.size(),Vector(0));
429  mCV.assign(mCells.size(),Scalar(0));
430  mReversed.assign(mFacets.size(),false);
431 
432  /* face centre: centroid of face vertices*/
433  forEach(mFacets,i) {
434  Facet& f = mFacets[i];
435  Vector C(0);
436  forEach(f,j)
437  C += mVertices[f[j]];
438  mFC[i] = C / Scalar(f.size());
439  }
440  /* cell centre: centroid of cell vertices */
441  forEach(mCells,i) {
442  Cell& c = mCells[i];
443  Vector C(0);
444  forEach(c,j)
445  C += mFC[c[j]];
446  mCC[i] = C / Scalar(c.size());
447  }
448  /* face normal */
449  forEach(mFacets,i) {
450  Facet& f = mFacets[i];
451  Vector N(0),C(0),Ci,Ni;
452  Scalar Ntot = Scalar(0);
453  const Vector& v1 = mFC[i];
454  forEach(f,j) {
455  const Vector& v2 = mVertices[f[j]];
456  const Vector& v3 = mVertices[f[ (j + 1 == f.size()) ? 0 : (j + 1) ]];
457 
458  Ni = ((v2 - v1) ^ (v3 - v1));
459  Scalar magN = mag(Ni);
460  Ci = magN * ((v1 + v2 + v3) / 3);
461 
462  C += Ci;
463  Ntot += magN;
464  N += Ni;
465  }
466  /*corrected face center: which is the centroid of the solid polygon
467  instead of the vertices alone*/
468  mFC[i] = C / Ntot;
469  Vector v = mFC[i] - mCC[mFOC[i]];
470  if((v & N) < 0) {
471  N = -N;
472  mReversed[i] = true;
473  }
474  mFN[i] = N / Scalar(2);
475  }
476  /* cell volumes */
477  for(i = 0;i < mBCS;i++) {
478  Cell& c = mCells[i];
479  Scalar V(0);
480  Vector C(0);
481  forEach(c,j) {
482  Vector v = mCC[i] - mFC[c[j]];
483  Scalar Vi = mag(v & mFN[c[j]]);
484  C += Vi * (3 * mFC[c[j]] + mCC[i]) / 4;
485  V += Vi;
486  }
487  /*corrected cell center: which is the centroid of the solid control volume
488  instead of the vertices alone*/
489  mCC[i] = C / V;
490  mCV[i] = V / Scalar(3);
491  }
492  /*correct for sphere*/
493  if(is_spherical) {
494  for(i = 0;i < mBCS;i++) {
495  Cell& c = mCells[i];
496  Scalar radiusb = mag(mVertices[mFacets[c[0]][0]]);
497  Scalar radiust = mag(mVertices[mFacets[c[1]][0]]);
498  mCC[i] = ((radiusb + radiust) / (2 * mag(mCC[i]))) * mCC[i];
499  mFC[c[0]] = (radiusb / mag(mFC[c[0]])) * mFC[c[0]];
500  mFC[c[1]] = (radiust / mag(mFC[c[1]])) * mFC[c[1]];
501  for(Int j = 2; j < 6; j++) {
502  mFC[c[j]] = ((radiusb + radiust) / (2 * mag(mFC[c[j]]))) * mFC[c[j]];
503  /*compute area of the four vertical faces of cubed-sphere elements*/
504  Scalar d = 0;
505  Facet& f = mFacets[c[j]];
506  forEach(f,k) {
507  Vector v0 = mVertices[f[k]];
508  Vector v1 = mVertices[f[ (k == f.size() - 1) ? 0 : k + 1]];
509  Vector r0 = unit(v0), r1 = unit(v1);
510  if(equal(r0,r1))
511  continue;
512  v0 = cart_to_sphere(v0);
513  v1 = cart_to_sphere(v1);
514  d += geodesic_distance(v0,v1);
515  }
516  d /= 2;
517  Scalar area = mag(radiust - radiusb) * d;
518  mFN[c[j]] = area * unit(mFN[c[j]]);
519  }
520  }
521  /*compute surface area and volume of cubed-sphere elements*/
522  for(i = 0;i < mBCS;i++) {
523  Cell& c = mCells[i];
524  Scalar area = 0;
525  for(Int k = 0; k < 2; k++) {
526  Scalar radius = mag(mVertices[mFacets[c[k]][0]]);
527  Facet& f = mFacets[c[k]];
528  Vector& fc = mFC[c[k]];
529  Scalar a = 0;
530  forEach(f,j) {
531  Vector& v0 = mVertices[f[j]];
532  Vector& v1 = mVertices[f[ (j == f.size() - 1) ? 0 : j + 1]];
533  a += spherical_triangle_area(radius,v0,v1,fc);
534  }
535  mFN[c[k]] = a * unit(mFN[c[k]]);
536  area += a;
537  }
538  area /= 2;
539  Scalar radiusb = mag(mVertices[mFacets[c[0]][0]]);
540  Scalar radiust = mag(mVertices[mFacets[c[1]][0]]);
541  mCV[i] = mag(radiust - radiusb) * area;
542  }
543  }
544  /*boundary cell centre and volume*/
545  forEachS(mCells,i,mBCS) {
546  Int fi = mCells[i][0];
547  mCV[i] = mCV[mFOC[fi]];
548  mCC[i] = mFC[fi];
549  }
